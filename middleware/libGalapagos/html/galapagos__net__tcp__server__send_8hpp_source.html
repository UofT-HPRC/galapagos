158 void tcp_server_send<T>::send_loop(){
159 
160  do{
161  if(!s_axis->empty()){
162  short dest = s_axis->get_head_dest();
163 #if LOG_LEVEL > 0
164  logger->debug("tcp_server_send, in send loop, sending buffer to dest{0:d}", dest);
165 #endif
166  std::string ip_addr = sessions->get_ip_addr(dest);
167  bool session_found = sessions->find(ip_addr);
168  if (!session_found){
169  send_new(ip_addr);
170  }
171  else{
172  sessions->get_s_axis(ip_addr)->splice(s_axis);
173 #if LOG_LEVEL > 0

175 #endif
176  }
177  }
178 
179  }while(!dc->is_done());
180 
181  dc->clean();
182 }
