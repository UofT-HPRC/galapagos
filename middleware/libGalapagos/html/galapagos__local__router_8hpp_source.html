208  logger->debug("in routing function, circling through {0:d} kernels", m_axis_ptr.size());
209 #endif
210 
211 
212  do{
213  for(unsigned int i=0; i<m_axis_ptr.size(); i++){
214  galapagos::interface <T> * _m_axis = m_axis_ptr[i];

217 #if LOG_LEVEL > 0
218  logger->debug("Node Routing Packet with dest:{0:x}", head_dest);
219  logger->debug("Node Routing to net address:{0}, my_address is {1}", kern_info_table[head_dest], my_address);
220 #endif
221 
222  if (kern_info_table[head_dest] == my_address)
223  {
224 #if LOG_LEVEL > 0
225  logger->debug("Node Routing Locally to index:{0:d}", dest_to_kern_ind[head_dest]);
226 #endif
227  s_axis_ptr[dest_to_kern_ind[head_dest]]->splice(_m_axis);
228  }//if (kern_info_table[head_dest] == my_address)
229  else
230  {//currently only external routing is to network
231 #if LOG_LEVEL > 0
232  logger->debug("Node routing to network address {0}", kern_info_table[head_dest]);
233  logger->flush();
234 #endif
235  std::lock_guard <std::mutex> guard(*in_flight_struct.mutex_packets_in_flight);
236  s_axis_ptr[num_local + NETWORK_EXT_INDEX]->splice(_m_axis);
237  (*(in_flight_struct.packets_in_flight))++;
238  }//else
239  }//if(!_m_axis->empty())
240  }//for
241 
242  }while(!dc->is_done());

244  dc->clean();
