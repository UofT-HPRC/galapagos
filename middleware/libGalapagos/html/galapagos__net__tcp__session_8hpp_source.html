262 void tcp_session<T>::do_read()
263 {
264  int length;
265  unsigned int num_read = 0;
266  int dest;
267  int id;
268  int size;
269 
270  do{
271 
272  socket.wait(boost::asio::ip::tcp::socket::wait_read);
273  unsigned int avail = socket.available();
274  if(avail>0){
275  boost::system::error_code error;
276  {
277  char data[(MAX_BUFFER+1)*sizeof(T)];
278 
279  T * header = (T *)data;
280  length = socket.read_some(boost::asio::buffer((char *)header, sizeof(T)), error);
281 
282  // dest = (int)header->range(31,24);
283  // id = (int)header->range(23,16);
284  // size = (int)header->range(15,0);
285  dest = galapagos::range(31, 24, *header);
286  id = galapagos::range(23, 16, *header);
287  size = galapagos::range(15, 0, *header);
288 
289  num_read=0;
290 #if LOG_LEVEL > 0
291  logger->debug ("do_read, size is {0:d}, max_size is {1:d}, avail is {2:d}", size, (MAX_BUFFER+1), avail);
292  logger->flush();
293 #endif
294  assert(size <= (MAX_BUFFER + 1));
295  while(num_read < size*sizeof(T)){
296  avail = socket.available();
297  if(avail > 0){
298  if(avail < (size*sizeof(T)-num_read)){
299  assert(avail <= (MAX_BUFFER + 1)*sizeof(T));
300  length = socket.read_some(boost::asio::buffer((char *)data + sizeof(T) + num_read, avail), error);
301  }
302  else{
303  assert((size*sizeof(T)-num_read) <= (MAX_BUFFER + 1)*sizeof(T));
304  length = socket.read_some(boost::asio::buffer((char *)data + sizeof(T) + num_read, size*sizeof(T)-num_read), error);
305  }
306  num_read +=length;
307  }
308  }
309 #if LOG_LEVEL > 0
310  logger->debug("Net:Received packet of {0:d} size at dest {1:x} from id {2:x}", size, dest, id);
311  logger->flush();
312 #endif
313  //assert( size % 8 == 0);
314  m_axis.packet_write(data + sizeof(T), size, dest, id);
315  }
316  }
317  }while(!dc->is_done());
318 
319  read_dc->clean();
